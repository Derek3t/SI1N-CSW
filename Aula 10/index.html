def celsius_para_fahrenheit(celsius):
    # Aplica a fórmula de conversão
    fahrenheit = celsius * 9 / 5 + 32
    return fahrenheit

# Solicita ao usuário que insira a temperatura em Celsius
temperatura_celsius = float(input("Digite a temperatura em graus Celsius: "))

# Converte a temperatura para Fahrenheit
temperatura_fahrenheit = celsius_para_fahrenheit(temperatura_celsius)

# Exibe o resultado
print(f"A temperatura de {temperatura_celsius}°C é equivalente a {temperatura_fahrenheit}°F.")
